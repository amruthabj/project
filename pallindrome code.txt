bool isValidWord(String str) {
  return str.contains(RegExp(r"^[a-zA-Z]+$"));
}

bool isPalindrome(String str) {
  str = str.toLowerCase();
  int start = 0;
  int end = str.length - 1;

  while (start < end) {
    if (str[start] != str[end]) {
      return false;
    }
    start++;
    end--;
  }
  return true;
}
